#### 父组件传参给子组件
>父组件 使用v-bind或省略 绑定数据
```
<cart-control :food="food"></cart-control>
```
>在子组件 使用props
export default {
    props:{
        food:{type:Object}
    }
}

#### 子组件传递参数给父组件
>在子组件使用$.emit()方法
```
addCart(event){
    //cartAdd父组件方法 event.target 传递的参数
    this.$emit('cartAdd',event.target);
},
```
>在父组件
```
//v-on:后面是子组件中this.$emit('cartAdd',event.target);中方法
<cart-control v-on:cartAdd="cartAdd"></cart-control>

cartAdd(target){
    this._drop(target)
}

```
#### 父组件调用子组件方法
>在父组件引用的子组件添加ref="shopcart"
```
<shop-cart ref="shopcart"></shop-cart>
 //drop是子组件方法,target参数
_drop(target){
    this.$refs.shopcart.drop(target)
},
```
>在子组件
```
drop(el){
    window.console.log("shopcart:"+el)
}

```
#### 父组件传值到子组件，子组件中props无法赋值到data中
解决：添加watch
```
props:{
    food:{
        type:Object
    }
},
data(){
    return {
        foods:this.food,
    }
},
watch:{//非常重要，不然为空
    food(obj){
        this.foods = obj;
    }
},
```
